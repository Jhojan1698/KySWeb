239         private void AnadirCarrito(HttpServletRequest request, HttpServletResponse response)
240                 throws ServletException, IOException {
241             
242             DecimalFormat df = new DecimalFormat("0.00");
243             DecimalFormatSymbols dfs = df.getDecimalFormatSymbols();
244             dfs.setDecimalSeparator('.');
245             df.setDecimalFormatSymbols(dfs);
246             
247             HttpSession session = request.getSession();
248             ArrayList<DetalleVenta> carritoCompra;
249             
250             if(session.getAttribute("carrito") == null){
251                 carritoCompra = new ArrayList<DetalleVenta>();
252             }else{
253                 carritoCompra = (ArrayList<DetalleVenta>)session.getAttribute("carrito");
254             }
255             
256             Producto p = Producto_DB.listarProductoPorCodigo(request.getParameter("txtCodigo")); 
257             
258             DetalleVenta dv = new DetalleVenta();
259             dv.setCodigoProducto(request.getParameter("txtCodigo")); 
260             dv.setNombreProducto(request.getParameter("txtNombreP")); 
261             dv.setPrecio(Double.parseDouble(request.getParameter("txtPrecio")));
262             dv.setCantidad(Double.parseDouble(request.getParameter("txtCantidad"))); 
263             
264             // Realizamos un descuento para el sub total
265             double subTotal = p.getPrecioP() * dv.getCantidad();
266             if(subTotal > 1000){
267                 dv.setDescuento(subTotal * 0.25);
268             }
269             else if(subTotal > 400){
270                 dv.setDescuento(subTotal * 0.10);
271             }
272             else{
273                 dv.setDescuento(0); 
274             }
275             
276             dv.setSubTotal(dv.getPrecio() * dv.getCantidad() - dv.getDescuento());
277             
278             int indice = -1;
279             for(int i=0; i<carritoCompra.size(); i++){
280                 DetalleVenta det = carritoCompra.get(i);
281                 if(det.getCodigoProducto().equals(p.getCodigoP())){
282                     indice = i;
283                     break;
284                 }
285             }
286             
287             if(indice == -1){
288                 dv.setNumero(String.valueOf(carritoCompra.size() + 1)); 
289                 carritoCompra.add(dv);
290             }else{
291                 dv.setNumero(String.valueOf(indice + 1));
292                 carritoCompra.set(indice, dv);
293             }
294             
295             session.setAttribute("carrito", carritoCompra); 
296             request.getSession().setAttribute("parametroCliente", request.getParameter("txtCliente"));
297             response.sendRedirect("RegistrarVenta.jsp");
298         }
