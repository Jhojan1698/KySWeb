14     public class DetalleVenta_DB {
15         public static ArrayList<DetalleVenta> obtenerDetalleVenta(String CodigoVenta){

17             
18             Connection cn;
19             Conexion con = new Conexion();
20             cn = con.conectar();

22                 CallableStatement cs = cn.prepareCall("CALL MOSTRAR_DETALLE_VENTA_POR_CODIGO (?)");

24                 ResultSet rs = cs.executeQuery();
25                 
26                 while(rs.next()){
27                     DetalleVenta dv = new DetalleVenta();

29                     dv.setNombreProducto(rs.getString("Producto"));
30                     dv.setPrecio(rs.getDouble("Precio"));
31                     dv.setCantidad(rs.getDouble("Cantidad"));
32                     dv.setDescuento(rs.getDouble("Descuento"));
33                     dv.setSubTotal(rs.getDouble("SubTotal")); 
34                     lista.add(dv);
35                 }
36             }catch(Exception e){System.out.println(e);}
37             return lista;
38         }
