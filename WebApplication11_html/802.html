86                 
87                 if("null".equals(CodigoVenta)){

89                 }
90                 
91                 for(int i=0; i<NombreProducto.length;i++){
92                     try{
93                         CallableStatement cs = cn.prepareCall("CALL REGISTRAR_DETALLE_VENTA (?,?,?,?,?,?)");
